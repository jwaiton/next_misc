  45def ratio_error(f, a, b):
  46    '''
  47    Error multiplication via quadrature
  48
  49    f - efficiency (%)
  50    a - events post-cut
  51    b - total events
  52    a_error - sqrt of events post-cut
  53    b_error - sqrt of total events
  54    '''
  55
  56    a_error = np.sqrt(a)
  57    b_error = np.sqrt(b)
  58
  59    f_error = f*np.sqrt((a_error/a)**2
  60                                  +(b_error/b)**2)
  61    return f_error

 266def remove_low_E_events(df, energy_limit = 0.05):
 267    '''
 268    Remove low energy tracks, add their energy back to the first
 269    track and then update 'numb_of_tracks' to be up to date
 270    '''
 271
 272    tracks_test = df.copy(deep=True)
 273
 274    # take events with lower than 50 keV, 0.05 MeV
 275    condition = (tracks_test.energy < energy_limit)
 276    summed_df = tracks_test[condition].groupby('event')['energy'].sum().reset_index()
 277
 278     # merge these as a new column
 279    merged_df = pd.merge(tracks_test, summed_df, on='event', suffixes=('', '_sum'), how = 'left').fillna(0)
 280    # add this summed energy to first column
 281    merged_df['energy'] = merged_df.apply(lambda row: (row['energy'] + row['energy_sum']) if row.name == merged_df[merged_df['event'] == row['event']].index[0] else row['energy'], axis=1)
 282
 283    # drop energy sum column
 284    result_df = merged_df.drop('energy_sum', axis = 1)
 285
 286    # then remove all tracks below the energy threshold
 287    condition_upper = (result_df.energy > energy_limit)
 288    remove_low_E = result_df[condition_upper]
 289
 290    # count the number of events identified with unique event, and change numb_of_tracks to reflect this
 291    event_counts = remove_low_E['event'].value_counts(sort = False)
 292
 293    # apply this to numb_of_tracks
 294    remove_low_E['numb_of_tracks'] = remove_low_E['event'].map(event_counts)
 295
 296    return remove_low_E
